Project Configuration File

The following is an example of the 'config.json' file. Each project will have one of these which the server read and maps the websites vhost preview in Management UI, proxy mapping and DNS FQDN names via environment, maintenance mode for given website, error pages, and a series of sections like API mapping (server side execute), static path content and server execute overrides. 

   {
      "project_desc": "Some description",
      "enabled": true,
      "proxy_map": {
         "dev": {},
         "qa": {},
         "stage": {},
         "prod": {}
      },
      "dns_names": {
         "dev": {
            "localtest.network.local": "website_name"
         },
         "qa": {},
         "stage": {},
         "prod": {}
      },
      "websites": {
         "website_name": {
            "ssl_redirect": true,
            "maintenance": {
               "dev": false,
               "qa": false,
               "stage": false,
               "prod": false
            },
            "maintenance_page": "maintenance.html",
            "maintenance_page_api": "maintenance.json",
            "default_doc": "index.html",
            "default_errors": {
               "user": {
                  "404": "404.html",
                  "500": "500.html"
               },
               "api": {
                  "404": "404.json",
                  "500": "500.json"
               }
            },
            "apis_fixed_path": {},
            "apis_dynamic_path": {
               "/api/": "/website_name/api/"
            },
            "path_static": {
               "/": "/website_name/"
            },
            "path_static_server_exec": {},
            "sub_map": {}
         }
      }
   }
